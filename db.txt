after insert on emp
for each row
begin
   insert into emp2 values(:new.eid,:new.fname,:new.dept);

create or replace function get_avg_sal(did number)
return number 
is 
avg_sal number;
begin
   select avg(sal) into avg_sal from emp where dept=did;
   return avg_sal;
end;


 select get_avg_sal(2) from dual;



create or replace function get_high_sal
return number 
is 
high_sal number;
begin
   select  max(sal) into  high_sal from emp ;
   return  high_sal;
end;

 select get_high_sal(2) from dual;




create or replace function check_sal_level(eid1 number)
return varchar(255) 
is 
sal1 number;
begin
    select sal as sal1 from emp where eid=eid1;
   if  sal1 >=60000  then
       return 'high sal';
    else
    return 'normal sal';
   end if;
end;


 select check_sal_level(1) from dual;
